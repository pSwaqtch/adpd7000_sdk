178int32_t adi_adpd7000_ecg_read_fifo(adi_adpd7000_device_t *device, adi_adpd7000_fifo_config_t *fifo, uint32_t *ecg_data, uint8_t *ecg_num)
179{
180 int32_t err;
181 uint8_t i, ecg_count = 0;;
182 uint8_t fifo_data[4];
183 ADPD7000_NULL_POINTER_RETURN(device);
184 ADPD7000_LOG_FUNC();
185 ADPD7000_NULL_POINTER_RETURN(fifo);
186 ADPD7000_NULL_POINTER_RETURN(ecg_data);
187 ADPD7000_NULL_POINTER_RETURN(ecg_num);
188
189 if (fifo->ecg_slot)
190 {
191 for (i = 0; i < fifo->ecg_over_sample; i++)
192 {
193 err = adi_adpd7000_device_fifo_read_bytes(device, fifo_data, fifo->ecg_size);
194 ADPD7000_ERROR_RETURN(err);
195 if ((fifo_data[0] == 0xff) && (fifo->ecg_size == 4) && ((fifo->ppg_slot != 0) || (fifo->bioz_slot != 0)))
196 {
197 continue;
198 }
199 else
200 {
201 if (fifo->ecg_size == 4)
202 {
203 *ecg_data++ = (fifo_data[0] << 24) | (fifo_data[1] << 16) | (fifo_data[2] << 8) | (fifo_data[3]);
204 }
205 else
206 {
207 *ecg_data++ = ((fifo_data[0] << 16) | (fifo_data[1] << 8) | (fifo_data[2]));
208 }
209 ecg_count++;
210 }
211 }
212 }
213 *ecg_num = ecg_count;
214
215 return API_ADPD7000_ERROR_OK;
