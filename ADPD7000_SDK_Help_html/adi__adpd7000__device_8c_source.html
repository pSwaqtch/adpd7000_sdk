189int32_t adi_adpd7000_device_get_sequence_fifo_config(adi_adpd7000_device_t *device, adi_adpd7000_fifo_config_t *fifo)
190{
191 int32_t err;
192 uint16_t data, i;
193 ADPD7000_NULL_POINTER_RETURN(device);
194 ADPD7000_LOG_FUNC();
195 ADPD7000_NULL_POINTER_RETURN(fifo);
196
197 err = adi_adpd7000_hal_bf_read(device, BF_ECG_TIMESLOT_EN_INFO, &data);
198 ADPD7000_ERROR_RETURN(err);
199 fifo->ecg_slot = data;
200 err = adi_adpd7000_hal_bf_read(device, BF_PPG_TIMESLOT_EN_INFO, &data);
201 ADPD7000_ERROR_RETURN(err);
202 fifo->ppg_slot = data;
203 err = adi_adpd7000_hal_bf_read(device, BF_BIOZ_TIMESLOT_EN_INFO, &data);
204 ADPD7000_ERROR_RETURN(err);
205 fifo->bioz_slot = data;
206
207 err = adi_adpd7000_hal_bf_read(device, BF_ENA_STAT_ECG_INFO, &data);
208 ADPD7000_ERROR_RETURN(err);
209 fifo->ecg_size = (data == 1) ? 4 : 3;
210 if ((fifo->ppg_slot == 0) && (fifo->bioz_slot == 0))
211 {
212 fifo->ecg_over_sample = 1;
213 }
214 else
215 {
216 err = adi_adpd7000_hal_bf_read(device, BF_ECG_OVERSAMPLING_RATIO_INFO, &data);
217 ADPD7000_ERROR_RETURN(err);
218 fifo->ecg_over_sample = data;
219 }
220 fifo->sequence_size = fifo->ecg_slot * fifo->ecg_over_sample * fifo->ecg_size;
221
222 fifo->ppg_chnl_num = 0;
223 for (i = 0; i < fifo->ppg_slot; i++)
224 {
225 fifo->ppg_chnl_num += 1;
226 err = adi_adpd7000_hal_bf_read(device, ADPD7000_TIME_SLOT_SPAN * i + BF_CHANNEL_EN_A_INFO, &data);
227 ADPD7000_ERROR_RETURN(err);
228 fifo->ppg_fifo[i].ppg_chl2_en = data;
229 fifo->ppg_chnl_num += fifo->ppg_fifo[i].ppg_chl2_en;
230
231 err = adi_adpd7000_hal_bf_read(device, ADPD7000_TIME_SLOT_SPAN * i + BF_SIGNAL_SIZE_A_INFO, &data);
232 ADPD7000_ERROR_RETURN(err);
233 fifo->ppg_fifo[i].signal_size = data;
234
235 err = adi_adpd7000_hal_bf_read(device, ADPD7000_TIME_SLOT_SPAN * i + BF_DARK_SIZE_A_INFO, &data);
236 ADPD7000_ERROR_RETURN(err);
237 fifo->ppg_fifo[i].dark_size = data;
238
239 err = adi_adpd7000_hal_bf_read(device, ADPD7000_TIME_SLOT_SPAN * i + BF_LIT_SIZE_A_INFO, &data);
240 ADPD7000_ERROR_RETURN(err);
241 fifo->ppg_fifo[i].lit_size = data;
242
243 fifo->sequence_size += (fifo->ppg_fifo[i].signal_size + fifo->ppg_fifo[i].dark_size +fifo->ppg_fifo[i].lit_size) * (1 + fifo->ppg_fifo[i].ppg_chl2_en);
244 }
245
246 fifo->sequence_size += 6 * fifo->bioz_slot;
247
248 return API_ADPD7000_ERROR_OK;
249}
