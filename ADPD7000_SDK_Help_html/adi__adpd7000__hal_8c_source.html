21 ADPD7000_NULL_POINTER_RETURN(device);
22 ADPD7000_NULL_POINTER_RETURN(reg_data);
23
24 address = (reg_addr << 1);
25 wr_buf[0] = ((address >> 8) & 0xFF); /* address [15:08] */
26 wr_buf[1] = ((address ) & 0xFF); /* address [07:00] */

58int32_t adi_adpd7000_hal_bf_read(adi_adpd7000_device_t *device, uint32_t reg_addr, uint32_t bf_info, uint16_t *bf_val)
59{
60 uint16_t reg_value, reg_mask;
61 int32_t err;
62 uint8_t bit_start = bf_info;
63 uint8_t bit_count = bf_info >> 8;
64 ADPD7000_NULL_POINTER_RETURN(device);
65 ADPD7000_NULL_POINTER_RETURN(bf_val);
66 ADPD7000_INVALID_PARAM_RETURN((bit_count > 16) || (bit_count == 0));
67
68 err = adi_adpd7000_hal_reg_read(device, reg_addr, &reg_value);
69 ADPD7000_ERROR_RETURN(err);
70 if((bit_count + bit_start) <= 16)
71 {
72 reg_mask = (bit_count == 16) ? 0xffff : ((1 << bit_count) - 1);
73 *bf_val = (reg_value >> bit_start) & reg_mask;
74 }
75 else
76 {
77 *bf_val = reg_value >> bit_start;
78 err = adi_adpd7000_hal_reg_read(device, reg_addr + 4, &reg_value);
79 ADPD7000_ERROR_RETURN(err);
80 reg_mask = (1 << (bit_count - (16 - bit_start))) - 1;
81 *bf_val += (reg_value & reg_mask) << (16 - bit_start);
82 }
83
84 return API_ADPD7000_ERROR_OK;
85}

87int32_t adi_adpd7000_hal_bf_write(adi_adpd7000_device_t *device, uint32_t reg_addr, uint32_t bf_info, uint16_t bf_val)
88{
89 uint16_t reg_value, reg_mask;
90 int32_t err;
91 uint8_t bit_start = bf_info;
92 uint8_t bit_count = bf_info >> 8;
93 ADPD7000_NULL_POINTER_RETURN(device);
94 ADPD7000_INVALID_PARAM_RETURN((bit_count > 16) || (bit_count == 0));
95
96 if ((bit_count == 16) && (bit_start == 0))
97 {
98 err = adi_adpd7000_hal_reg_write(device, reg_addr, bf_val);
99 ADPD7000_ERROR_RETURN(err);
100 }
101 else
102 {
103 err = adi_adpd7000_hal_reg_read(device, reg_addr, &reg_value);
104 ADPD7000_ERROR_RETURN(err);
105 if((bit_count + bit_start) <= 16)
106 {
107 reg_mask = (bit_count == 16) ? 0xffff : ((1 << bit_count) - 1);
108 reg_value &= ~(reg_mask << bit_start);
109 reg_value |= (bf_val << bit_start);
110 err = adi_adpd7000_hal_reg_write(device, reg_addr, reg_value);
111 ADPD7000_ERROR_RETURN(err);
112 }
113 else
114 {
115 reg_mask = 0xffff;
116 reg_value &= ~(reg_mask << bit_start);
117 reg_value |= (bf_val << bit_start);
118 err = adi_adpd7000_hal_reg_write(device, reg_addr, reg_value);
119 ADPD7000_ERROR_RETURN(err);
120 err = adi_adpd7000_hal_reg_read(device, reg_addr + 4, &reg_value);
121 ADPD7000_ERROR_RETURN(err);
122 reg_mask = (1 << (bit_count - (16 - bit_start))) - 1;
123 reg_value &= ~reg_mask;
124 reg_value |= (bf_val >> (16 - bit_start));
125 err = adi_adpd7000_hal_reg_write(device, reg_addr, reg_value);
126 ADPD7000_ERROR_RETURN(err);
127 }
128 }
129
130 return API_ADPD7000_ERROR_OK;
