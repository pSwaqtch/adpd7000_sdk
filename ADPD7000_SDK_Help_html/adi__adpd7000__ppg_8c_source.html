50{
51 int32_t err;
52 uint8_t i, slot_num, channel;
53 uint16_t slot_chnl_num;
54 ADPD7000_NULL_POINTER_RETURN(device);
55 ADPD7000_LOG_FUNC();
56
57 err = adi_adpd7000_ppg_get_slot_num(device, &slot_num);
58 ADPD7000_ERROR_RETURN(err);
59
60 channel = 0;
61 for (i = 0; i < slot_num; i++)
62 {
63 err = adi_adpd7000_hal_bf_read(device, ADPD7000_TIME_SLOT_SPAN * i + BF_CHANNEL_EN_A_INFO, &slot_chnl_num);
64 ADPD7000_ERROR_RETURN(err);
65 channel = channel + slot_chnl_num + 1;
66 }
67 *channel_num = channel;
68
69 return API_ADPD7000_ERROR_OK;

672int32_t adi_adpd7000_ppg_agc_process(adi_adpd7000_device_t *device, adi_adpd7000_fifo_config_t *fifo, adi_adpd7000_ppg_agc_cfg_t *ppg_agc_cfg, uint32_t *signal_data)
673{
674 int32_t err;
675 uint8_t i;
676 uint32_t data;
677 float step;
678 float ppg_data_avg, ppg_chnl_current;
679 ADPD7000_NULL_POINTER_RETURN(device);
680 ADPD7000_LOG_FUNC();
681 ADPD7000_NULL_POINTER_RETURN(fifo);
682 ADPD7000_NULL_POINTER_RETURN(ppg_agc_cfg);
683 ADPD7000_NULL_POINTER_RETURN(signal_data);
684
685 ppg_sample_count++;
686 if (ppg_sample_count <= ppg_agc_cfg->ppg_skip_sample_number)
687 {
688 return API_ADPD7000_ERROR_OK;
689 }
690 if (ppg_sample_count <= (ppg_agc_cfg->ppg_skip_sample_number + ppg_agc_cfg->ppg_average_sample_number))
691 {
692 for (i = 0; i < fifo->ppg_slot; i++)
693 {
694 if (fifo->ppg_fifo[i].ppg_chl2_en == 0)
695 {
696 ppg_agc_run[i].ppg_data_sum += *signal_data++;
697 }
698 else
699 {
700 if (ppg_agc_cfg->slot[i].tia_chnl == 0)
701 {
702 ppg_agc_run[i].ppg_data_sum += *signal_data++;
703 data = *signal_data++;
704 }
705 else if (ppg_agc_cfg->slot[i].tia_chnl == 1)
706 {
707 data = *signal_data++;
708 ppg_agc_run[i].ppg_data_sum += *signal_data++;
709 }
710 }
711 }
712 data = data;
713 return API_ADPD7000_ERROR_OK;
714 }
715
716 for (i = 0; i < fifo->ppg_slot; i++)
717 {
718 if ((ppg_agc_cfg->slot[i].agc_en == 0) || (ppg_agc_run[i].agc_done == 1))
719 {
720 continue;
721 }
722 ppg_data_avg = ppg_agc_run[i].ppg_data_sum / ppg_agc_cfg->ppg_average_sample_number;
723 ppg_data_avg ++;
724 ppg_chnl_current = (float)(ppg_agc_run[i].led_current * ppg_agc_run[i].ppg_full_scale * 0.5 / ppg_data_avg);
725 if (ppg_chnl_current > 127)
726 {
727 if(ppg_agc_run[i].tia_gain > 0)
728 {
729 ppg_agc_run[i].tia_gain = ppg_agc_run[i].tia_gain - 1;
730 ppg_chnl_current = ppg_agc_run[i].led_current;
731 }
732 else
733 {
734 ppg_chnl_current = ppg_agc_run[i].led_current + 0x10;
735 ppg_chnl_current = (ppg_chnl_current > 127) ? 127 : ppg_chnl_current;
736 }
737 }
738 else if (ppg_chnl_current < 1)
739 {
740 if(ppg_agc_run[i].tia_gain < 5)
741 {
742 ppg_agc_run[i].tia_gain = ppg_agc_run[i].tia_gain + 1;
743 ppg_chnl_current = ppg_agc_run[i].led_current;
744 }
745 else
746 {
747 ppg_chnl_current = (ppg_agc_run[i].led_current > 0x11) ? (ppg_agc_run[i].led_current - 0x10) : 1;
748 }
749 }
750
751 if (ppg_agc_cfg->power_first_en == 0) /* low tia gain */
752 {
753 if (fabs(ppg_agc_run[i].led_current - (uint8_t)ppg_chnl_current) > 1)
754 {
755 ppg_agc_run[i].led_current = (uint8_t)ppg_chnl_current;
756 }
757 else
758 {
759 if (ppg_agc_cfg->slot[i].agc_type == 1)
760 {
761 ppg_agc_run[i].agc_done = 1;
762 }
763 }
764
765 if ((ppg_agc_run[i].led_current < 0x20) && (ppg_agc_run[i].tia_gain < 5))
766 {
767 ppg_agc_run[i].led_current = ppg_agc_run[i].led_current * 2;
768 ppg_agc_run[i].tia_gain = ppg_agc_run[i].tia_gain + 1;
769 }
770 }
771 else /* low LED current */
772 {
773 step = fabs(ppg_chnl_current - ppg_agc_run[i].led_current) / 4;
774 if (step < 1)
775 {
776 step = 1;
777 }
778 if (ppg_agc_run[i].led_current > ((uint8_t)ppg_chnl_current + 1))
779 {
780 ppg_agc_run[i].led_current -= (uint8_t)step;
781 }
782 else if (ppg_agc_run[i].led_current < ((uint8_t)ppg_chnl_current - 1))
783 {
784 ppg_agc_run[i].led_current += (uint8_t)step;
785 }
786 if ((ppg_agc_run[i].led_current > 0x10) && (ppg_agc_run[i].tia_gain > 2))
787 {
788 ppg_agc_run[i].led_current = ppg_agc_run[i].led_current / 2;
789 ppg_agc_run[i].tia_gain = ppg_agc_run[i].tia_gain - 1;
790 }
791 }
792 }
793
794 adi_adpd7000_device_enable_slot_operation_mode_go(device, false);
795 for (i = 0; i < fifo->ppg_slot; i++)
796 {
797 if (ppg_agc_cfg->slot[i].agc_en == 1)
798 {
799 err = adi_adpd7000_ppg_tia_set_gain_res(device, i, ppg_agc_cfg->slot[i].tia_chnl, ppg_agc_run[i].tia_gain);
800 ADPD7000_ERROR_RETURN(err);
801 err = adi_adpd7000_ppg_led_set_current(device, i, ppg_agc_cfg->slot[i].led_chnl, ppg_agc_run[i].led_current);
802 ADPD7000_ERROR_RETURN(err);
803 }
804 }
805 adi_adpd7000_device_clr_fifo(device);
806 adi_adpd7000_device_enable_slot_operation_mode_go(device, true);
807
808 ppg_sample_count = 0;
809 for (i = 0; i < fifo->ppg_slot; i++)
810 {
811 ppg_agc_run[i].ppg_data_sum = 0;;
812 }
813
814 return err;
815}
